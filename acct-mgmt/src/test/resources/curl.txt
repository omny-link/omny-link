#
# GLOBAL
#

# LIST GLOBAL CONFIG
curl -H "Accept:application/json" -u tstephen:tstephen http://localhost:8082/configuration/

# LIST TENANT METRICS (SNAPHOT)
curl -H "Accept:application/json" -u tstephen:tstephen http://localhost:8082/admin/tenants/

# RECORD TENANT METRICS
curl -X POST -H "Accept:application/json" -u tstephen:tstephen http://localhost:8082/admin/metrics/

# LIST TENANT METRICS (HISTORIC)
curl -H "Accept:application/json" -u tstephen:tstephen http://localhost:8082/admin/metrics/

#
# PER TENANT
#

# CREATE TENANT
curl -H "Accept:application/json" -H "Content-Type: application/json" -X POST -d '{ "tenantId":"omny" }' -u tstephen:tstephen http://localhost:8082/tenantConfigs

# 
curl -H "Accept:application/json" -u tstephen:tstephen http://localhost:8082/admin/tenants/omny

# CREATE TENANT BOT 
curl -H "Accept:application/json" -X POST -u tstephen:tstephen http://localhost:8083/tenants/sdu/bot

# FORCE RECREATE BOT
 curl -v -H "Accept:application/json" -X POST -u tstephen:tstephen http://localhost:8083/tenants/sdu/bot?force=true
